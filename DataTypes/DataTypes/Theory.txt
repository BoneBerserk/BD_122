//Переменная (Variable) - это именованная область памяти,
//содержимое которой может изменяться в процессе выполнения программы.
//				Синтаксис объявления переменной:
//						type name;

						ТИПЫ ДАННЫХ
1. Логические типы: bool (Boolean) true либо false
	true  - 1;
	false - 0;
	0 - это false, а true - это все что НЕ 0!

2. Символные типы: char (Character - Символ) ASCII
	Всего ASCII-символов 256;

3. Числовые типы: 
	unsigned - беззнаковые
	  signed - знаковые
	-------------------------
	short, long, int, long long;
	int a;	//она сможет хранить как положительные, так и отрицательные целые числа.
	//Чтобы объявить беззнаковую переменную, перед ее объявлением нужно написать 
	//ключевое слово unsigned:
	unsigned int b;

	SHORT
	2B
	unsigned short: 0 ... 65 535;

	  signed short: -32 768 ... 32 767;
